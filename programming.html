# Asynchronous Python Example
import asyncio

async def fetch_data():
    print("Start fetching")
    await asyncio.sleep(2)
    print("Done fetching")
    return {"data": 1}

async def main():
    task = asyncio.create_task(fetch_data())
    await task

asyncio.run(main())
